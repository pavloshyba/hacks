#mount vbox shared folder

##### GIT #####

# file list in commit 
git diff-tree --no-commit-id --name-only -r bd61ad98

#deleted in stage
git ls-files --deleted -z | xargs -0 git rm 

# reset git file change mode
git diff -p \
    | grep -E '^(diff|old mode|new mode)' \
    | sed -e 's/^old/NEW/;s/^new/old/;s/^NEW/new/' \
    | git apply
